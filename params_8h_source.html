242  {
243  bool paramHelp = false;
244  size_t count = 0;
245  for (int i = 1; i < argc; i++) {
246  std::string param_str = to_string(argv[i]);
247  if (!isParam(param_str)) {
248  continue;
249  }
250  bool found = false;
251  param_str = param_str.substr(1); // skip the first char
252 
253  std::map<std::string, Param*>::iterator itr;
254  for (itr = myParams.begin(); itr != myParams.end(); ++itr) {
255  Param *param = itr->second;
256  if (param_str == PARAM_HELP2 || param_str == PARAM_HELP1)
257  {
258  this->info(false);
259  return false;
260  }
261 
262  if (param_str == param->argStr) {
263  // has argument:
264  if ((i + 1) < argc && !(isParam(to_string(argv[i + 1])))) {
265  std::string nextVal = to_string(argv[i + 1]);
266  //help requested explicitly
267  if (nextVal == PARAM_HELP1) {
268  paramHelp = true;
269  paramkit::print_in_color(RED, param_str);
270  param->printDesc();
271  found = true;
272  break;
273  }
274  param->parse(argv[i + 1]);
275  found = true;
276 #ifdef _DEBUG
277  std::cout << argv[i] << " : " << argv[i + 1] << "\n";
278 #endif
279  break;
280  }
281  // does not require an argument:
282  if (!param->requiredArg) {
283  param->parse((char*)nullptr);
284  found = true;
285  break;
286  }
287  // requires an argument, but it is missing:
288  paramkit::print_in_color(RED, param_str);
289  paramHelp = true;
290  param->printDesc();
291  found = true;
292  break;
293  }
294  }
295  if (found) {
296  count++;
297  }
298  else {
299  const std::string param_str = to_string(argv[i]);
300  printUnknownParam(param_str);
301  print_in_color(HILIGHTED_COLOR, "Similar parameters:\n\n");
302  this->info(false, param_str);
303  return false;
304  }
305  }
306  if (paramHelp) {
307  return false;

309  if (!this->hasRequiredFilled()) {
